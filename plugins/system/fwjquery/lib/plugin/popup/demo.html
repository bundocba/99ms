}

     alert('This message is call from Button No');
}

     fwconfirm('Hello! Please choose one option!', callOnYes, callOnNo);

Custom Popup
jQuery.fwpopup({
     message: 'Put your message here', // Message to show(HTML or Text)
     background: 'default: #000', //Allow CSS background attribute
     popupBackground: 'default: #000', //Allow CSS background attribute
     popupBorder: 'default: 1px solid red', //Allow CSS background attribute
     popupWidth: '250', // Width of popup window. Default 250px
     popupHeight: '200',// Width of popup window. Default 200px
     top: 0,// Top Position of the popup window
     backgroundClickClose: false, // Close on background Click
     speed: 300, // Popup window fadeIn speed
     opacity: 0.5, // The opacity of popup background
     closeButton: false, //Show Close button
     closeButtonClass: 'closeButton', //add class to config your close button
     buttonClass: 'button', //add class to config your button
     button: {
          my_first_button: {
               'value' : 'First Button',
               'onClick' : function() {
                    myFirstFunction();
               }
          },
          my_second_button: {
               'value' : 'Second Button',
               'onClick' : function() {
                    mySecondFunction();
               }
          },
     }, // allow you to add button
     iframe: {
          link: http://example.com,
          width: 350,
          height: 250
     } // your Iframe config. If you using iframe, message property will be auto disable
}
